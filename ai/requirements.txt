llama-index-llms-openai  # драйвер для LLM-моделей в llama-index (можно заменить на другой)

python-dotenv      # управление переменными окружения (.env)

httpx              # HTTP-клиент для запросов к внешним API (включая GitLab)

numpy              # базовая математика для обработки данных

pydub              # конвертация аудио (если понадобится при работе с голосом)
